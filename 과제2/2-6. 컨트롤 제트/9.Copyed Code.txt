#include <string>
#include <vector>
#include <sstream>

using namespace std;

int solution(string s) {
    int answer = 0;
    stringstream sstream;
    sstream.str(s);
 
    // 처음 입력은 무조건 answer로 넣어주고,
    // 해당 값을 이전 값에도 입력해놓는다
    sstream >> answer;
    int num = answer;
    
    // 들어온 값을 temp에 저장
    string temp;
    while (sstream >> temp)
    {
        // 만약 "Z"라면 이전값을 빼준다
        if (temp == "Z")
        {
            answer -= num;
        }
        // 이전값 갱신 + 현재값 더해주기
        else 
        {
            int temp_num = stoi(temp);
            answer += temp_num;
            num = temp_num;
        }
    }    
    return answer;
}
